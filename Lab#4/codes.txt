#include <stdio.h>
#include <stdlib.h>

int main()
{
    int percent;
    printf("enter percentage ");
    scanf("%d",&percent);

        if (percent<55)
            printf("fail");
        else if(percent>=55 && percent<65)
            printf("c+");
        else if(percent>=65 && percent<75)
            printf("B");
        else if(percent>=75 && percent<85)
            printf("B+");
        else if(percent>=85 && percent<95)
            printf("A");
        else if(percent>=95 && percent<=100)
            printf("A+");
        else
            printf("invalid");
}
